        def batch_create(sql, table)
                puts "Creating table \"#{table}\" with \"#{sql}\""
                begin
                        @db.execute_batch(sql)
                rescue SQLite3::SQLException
                        puts "Error creating table #{table}: SQL error"
                        false
                rescue
                        puts "Error creating table #{table}"
                        false
                end
        end
